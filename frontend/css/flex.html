# 基本布局，父元素高度自适应

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>flaxe基本布局</title>
    <style>
        .content {
            display: flex;
            flex-direction: row;
            background: red;
            flex-wrap: nowrap;
            justify-content: space-between;
            align-items: stretch;

        }

        .leftbox {
            width: 300px;
            background: blue;
        }

        .rightbox {
            width: 100%;
            min-height: 500px;
            background: green;
        }
    </style>
</head>

<body>
    <div class="content">
        <div class="leftbox">1</div>
        <div class="rightbox">2
            <div class="main">
                <textarea name="" id="" cols="30" rows="10"></textarea>
            </div>
        </div>
    </div>

</body>

</html>
